#Arrest the Kostusev(Суд Гриши Костусева)
country_event = {
	id = blr.1
	title = blr.1.t
    desc = blr.1.desc

    fire_only_once = yes

	trigger = {
		tag = BLR
		NOT = {
			has_country_flag = kostusev_in_prison
		}
	}
	
	option = {
		name = blr.1.a
		remove_country_leader_role = {
			character = kostusev
		}
		add_country_leader_role = {
			character = saranchuk
			promote_leader = yes
			country_leader = {
				ideology = nationalism
				traits = {  }
			}
		}

		set_country_flag = {
			flag = save_kostusev
			days = 3600
			value = 2
		}
	}
}
